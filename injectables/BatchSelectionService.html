projects/congacommerce/elements/src/shared/services/batch-selection.service.ts

Description

Btach selecion service provides methods to perform operation on selected product in the drawer.

Usage

Example :
import { BatchSelectionService, AObjectService} from '@congacommerce/ecommerce';

export class MyComponent implements OnInit{
constructor( private batchSelectionService: BatchSelectionService)
}
// or
export class MyService extends AObjectService {
private batchSelectionService: BatchSelectionService = this.injector.get(BatchSelectionService);
}

Public getSelectedLineItems
getSelectedLineItems()

Gets the list of currently selected lineItem.

Example:

Example :
import { BatchSelectionService, Product } from '@congacommerce/ecommerce';
import { Observable } from 'rxjs/Observable';

export class MyComponent implements OnInit{
lineItems: CartItem;
lineItems$: Observable<Array<CartItem>>;
constructor(private batchSelectionService: BatchSelectionService){}

getSelectedLineItems(){
this.batchSelectionService.getSelectedLineItems().subscribe((res)=> this.lineItems = res);
// or
this.lineItems$ = this.batchSelectionService.getSelectedLineItems();
}
}
Returns : Observable<Array<CartItem>>

Observable<Array> return an array list of cart items.
