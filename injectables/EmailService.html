Description

The email service provides methods for interacting with the emails of the current user has access to.

Usage

Example :
import { EmailService, AObjectService} from '@congacommerce/ecommerce';

export class MyComponent implements OnInit{
constructor( private emailService: EmailService)
}
// or
export class MyService extends AObjectService {
private emailService: EmailService = this.injector.get(EmailService);
}

Public guestUserNewOrderNotification
guestUserNewOrderNotification(orderId: string, orderURL: string)

Send email notification for new order to the guest user

Example:

Example :
import { EmailService } from '@congacommerce/ecommerce';
import { Observable } from 'rxjs/Observable';

export class MyComponent implements OnInit{
result;
constructor(private emailService: EmailService){}

guestUserNewOrderNotification(orderId: string, orderURL: string){
this.emailService.guestUserNewOrderNotification(orderId, orderUrl).subscribe(res=> this.result=res);
}
}
